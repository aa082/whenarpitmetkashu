<script>

function nextPage(pageNumber) { document.querySelector('.page.active').classList.remove('active'); document.getElementById('page' + pageNumber).classList.add('active');

}


function moveButton() {

const btn = document.getElementById('noBtn');

const scrapbook = document.querySelector('.scrapbook'); const maxX = scrapbook.clientWidth - btn.clientWidth - 50; const maxY = scrapbook.clientHeight - btn.clientHeight - 50;


btn.style.position = 'absolute';

btn.style.left = Math.random() * maxX + 'px'; btn.style.top = Math.random() * maxY + 'px';

}


function handleYes() { document.querySelector('.page.active').classList.remove('active');

document.getElementById('success').classList.add('active'); createHearts();

}


function createHearts() { setInterval(() => {

const heart = document.createElement('div'); heart.innerHTML = '

';

heart.style.position = 'fixed';

heart.style.fontSize = Math.random() * 30 + 20 + 'px'; heart.style.left = Math.random() * 100 + 'vw'; heart.style.top = '100vh';

heart.style.opacity = '0.8'; heart.style.transition = 'all 3s ease-out'; document.body.appendChild(heart);


setTimeout(() => { heart.style.top = '0vh'; heart.style.opacity = '0';

}, 100);


setTimeout(() => { document.body.removeChild(heart);

}, 3000);

}, 300);

}
